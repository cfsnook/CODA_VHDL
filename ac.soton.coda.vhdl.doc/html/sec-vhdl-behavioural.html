2.1 Behavioural VHDL

We use a subset of the behavioural VHDL syntax as defined in the VHDL standard 1076.

DesignFile

::=

DesignUnit+

DesignUnit

::=

ContextItem* LibraryUnit

ContextItem

::=

LibraryClause | UseClause

LibraryClause

::=

LIBRARY LogicalName (, LogicalName)*;

LogicalName

::=

Identifier

UseClause

::=

USE SelectedName (, SelectedName)*;

Name

::=

SelectedName |

LibraryUnit

::=

PrimaryUnit | SecondaryUnit

PrimaryUnit

::=

EntityDeclaration

EntityDeclaration

::=

ENTITY Identifier IS
   

  PORT (

   

     InterfaceList

   

  )

   

  EntityDeclarativeItem*

   

END Identifier

InterfaceList

::=

InterfaceElement (; InterfaceElement)*

InterfaceElement

::=

InterfaceSignalDeclaration |

InterfaceSignalDeclaration

::=

[SIGNAL] Identifier : [Mode] SubtypeIndication

Mode

::=

IN | OUT | INOUT

SubtypeIndication

::=

Name

EntityDeclarativeItem

::=

TypeDeclaration

TypeDeclaration

::=

FullTypeDeclaration

FullTypeDeclaration

::=

TYPE Identifier IS TypeDefinition ;

TypeDefinition

::=

ScalarTypeDefinition

ScalarTypeDefinition

::=

EnumerationTypeDefinition

EnumerationTypeDefinition

::=

( EnumerationLiteral (, EnumerationLiteral)* )

EnumeartionLiteral

::=

Name

SecondaryUnit

::=

ArchitectureBody

ArchitectureBody

::=

ARCHITECTURE Identifier OF Name IS
   

  BlockDeclarativeItem*

   

BEGIN
   

  ConcurrentStatement*

   

END Name ;

BlockDeclarativeItem

::=

TypeDeclaration

ConcurrentStatement

::=

ProcessStatement

ProcessStatement

::=

PROCESS [( SignalName* )] IS
   

  ProcessDeclarativeItem*

   

BEGIN
   

  SequentialStatement*

   

END PROCESS

SignalName

::=

Name

ProcessDeclarativeItem

::=

TypeDeclaration

SequentialStatement

AssertionStatement |

   

SignalAssignmentStatement |

AssertionStatement

ASSERT Assertion

Assertion

::=

BooleanExpression

SignalAssignmentStatement

::=

Target <– Waveform

IfStatement

::=

IfClause

   

ElsifClause*

   

[ElseClause]

IfClause

::=

IF BooleanExpression THEN
   

  SequentialStatement*

ElsifClause

::=

ELSIF BooleanExpression THEN
   

  SequentialStatement*

ElseClause

::=

ELSE
   

  SequentialStatement*

CaseStatement

::=

CASE Expression IS
   

  CaseStatementAlternative+

   

END CASE

CaseStatementAlternative

::=

WHEN Label =>
   

  SequentialStatement*
